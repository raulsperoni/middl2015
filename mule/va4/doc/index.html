EpuertoColocarOrden

Show configuration XML code of the flow EpuertoColocarOrden... 
<?xml version="1.0" encoding="UTF-8"?>

<sub-flow name="EpuertoColocarOrden" doc:description="Se procesa Colocar Orden de Epuerto">
    <set-variable variableName="identificadorCompra" value="#[xpath3('//ser:colocarOrden/identificadorCompra')]" doc:name="idCompra" doc:description="Guardo el identificador de compra"/>
    <data-mapper:transform config-ref="XML_To_CSV" doc:name="XML To CSV" doc:description="Paso de XML a CSV">   </data-mapper:transform>
    <file:outbound-endpoint path="target/recepcion" outputPattern="#[flowVars.identificadorCompra].csv" responseTimeout="10000" doc:name="Recepcion" doc:description="Guardo en carpeta Recepcion"> </file:outbound-endpoint>
    <data-mapper:transform config-ref="CSV_To_Pojo_1" doc:name="CSV To Pojo" doc:description="Paso a Objeto Java">  </data-mapper:transform>
    <component class="va4.EpuertoConfirmacion" doc:name="Java" doc:description="Bean que procesa la solicitud"/>
    <set-variable variableName="identificadorReserva" value="#[payload.identificadorReserva]" doc:name="idReserva" doc:description="Guardo el identificador de la Reserva"/>
    <data-mapper:transform config-ref="Pojo_To_CSV" doc:name="Pojo To CSV" doc:description="Convierto confirmacion de reserva en CSV">  </data-mapper:transform>
    <file:outbound-endpoint path="target/confirmacion" outputPattern="#[flowVars.identificadorReserva].csv" connector-ref="File" responseTimeout="10000" doc:name="Confirmacion" doc:description="Guardo en carpeta Confirmacion">  </file:outbound-endpoint>
    <data-mapper:transform config-ref="CSV_To_Pojo_1CSV_To_Pojo_2CSV_To_Pojo_3" doc:name="CSV To Pojo" doc:description="Paso a Objeto nuevamente para la respuesta">    </data-mapper:transform>
</sub-flow>

Se procesa Colocar Orden de Epuerto
